77template<typename T, typename I>
78VEG_NO_INLINE void
79noalias_symhiv_add_impl( //
80 VectorViewMut<T> out,
81 proxsuite::linalg::sparse::MatRef<T, I> a,
82 VectorView<T> in)
83{
84 VEG_ASSERT_ALL_OF /* NOLINT */ ( //
85 a.nrows() == a.ncols(),
86 a.nrows() == out.dim,
87 a.ncols() == in.dim);
88 // equivalent to
89 // out.to_eigen().noalias() +=
90 // a.to_eigen().template selfadjointView<Eigen::Upper>() *
91 // in.to_eigen();
92
93 auto* ai = a.row_indices();
94 auto* ax = a.values();
95 auto n = a.ncols();
96
97 for (usize j = 0; j < usize(n); ++j) {
98 usize col_start = a.col_start(j);
99 usize col_end = a.col_end(j);
100
101 if (col_start == col_end) {
102 continue;
103 }
104
105 T acc0 = 0;
106 T acc1 = 0;
107 T acc2 = 0;
108 T acc3 = 0;
109
110 T in_j = in(isize(j));
111
112 usize pcount = col_end - col_start;
113
114 auto zx = proxsuite::linalg::sparse::util::zero_extend;
115
116 if (zx(ai[col_end - 1]) == j) {
117 T ajj = ax[col_end - 1];
118 out(isize(j)) += ajj * in_j;
119 pcount -= 1;
120 }
121
122 usize p = col_start;
123
124 for (; p < col_start + pcount / 4 * 4; p += 4) {
125 auto i0 = isize(zx(ai[p + 0]));
126 auto i1 = isize(zx(ai[p + 1]));
127 auto i2 = isize(zx(ai[p + 2]));
128 auto i3 = isize(zx(ai[p + 3]));
129
130 T ai0j = ax[p + 0];
131 T ai1j = ax[p + 1];
132 T ai2j = ax[p + 2];
133 T ai3j = ax[p + 3];
134
135 out(i0) += ai0j * in_j;
136 out(i1) += ai1j * in_j;
137 out(i2) += ai2j * in_j;
138 out(i3) += ai3j * in_j;
139
140 acc0 += ai0j * in(i0);
141 acc1 += ai1j * in(i1);
142 acc2 += ai2j * in(i2);
143 acc3 += ai3j * in(i3);
144 }
145 for (; p < col_start + pcount; ++p) {
146 auto i = isize(zx(ai[p]));
147
148 T aij = ax[p];
149 out(i) += aij * in_j;
150 acc0 += aij * in(i);
151 }
152 acc0 = ((acc0 + acc1) + (acc2 + acc3));
153 out(isize(j)) += acc0;
154 }
155}

568template<typename T, typename I>
569void
570refactorize(SolverState<T> const& state,
571 proxsuite::linalg::sparse::MatRef<T, I> const& kkt_active,
572 bool const* active_constraints,
573 Workspace<T, I>& work,
574 proxsuite::linalg::veg::dynstack::DynStackMut stack)
575{
576 auto& aug_kkt = *work.internal.matrix_free_kkt.get();
577 auto& iterative_solver = *work.internal.matrix_free_solver.get();
578
579 isize n = aug_kkt.internal.n;
580 isize n_eq = aug_kkt.internal.n_eq;
581 isize n_in = aug_kkt.internal.n_in;
582 isize n_tot = n + n_eq + n_in;
583
584 if (work.internal.do_ldlt) {
585 Ldlt<T, I>& ldlt = work.internal.ldlt;
586 proxsuite::linalg::sparse::factorize_symbolic_non_zeros(
587 ldlt.nnz_counts.ptr_mut(),
588 ldlt.etree.ptr_mut(),
589 ldlt.perm_inv.ptr_mut(),
590 ldlt.perm.ptr(),
591 kkt_active.symbolic(),
592 stack);
593
594 LDLT_TEMP_VEC_UNINIT(T, diag, n_tot, stack);
595 for (isize i = 0; i < n; ++i) {
596 diag[i] = state.rho;
597 }
598 for (isize i = 0; i < n_eq; ++i) {
599 diag[n + i] = -state.mu_eq;
600 }
601 for (isize i = 0; i < n_in; ++i) {
602 diag[n + n_eq + i] = active_constraints[i] ? -state.mu_in : T(1);
603 }
604
605 proxsuite::linalg::sparse::factorize_numeric(ldlt.values.ptr_mut(),
606 ldlt.row_indices.ptr_mut(),
607 diag.data(),
608 ldlt.perm.ptr(),
609 ldlt.col_ptrs.ptr(),
610 ldlt.etree.ptr(),
611 ldlt.perm_inv.ptr(),
612 kkt_active,
613 stack);
614
615 } else {
616 aug_kkt = AugmentedKkt<T, I>{ {
617 kkt_active,
618 active_constraints,
619 n,
620 n_eq,
621 n_in,
622 state.rho,
623 state.mu_eq,
624 state.mu_in,
625 } };
626 iterative_solver.compute(aug_kkt);
627 }

638template<typename T, typename I>
639void
640ldl_solve_in_place(VectorViewMut<T> rhs,
641 Workspace<T, I> const& work,
642 proxsuite::linalg::veg::dynstack::DynStackMut stack)
643{
644 auto zx = proxsuite::linalg::sparse::util::zero_extend;
645
646 auto rhs_e = rhs.to_eigen();
647 auto n_tot = rhs_e.rows();
648 LDLT_TEMP_VEC_UNINIT(T, tmp, n_tot, stack);
649
650 if (work.internal.do_ldlt) {
651
652 Ldlt<T, I>& ldlt = work.internal.ldlt;
653
654 proxsuite::linalg::sparse::MatRef<T, I> ld = {
655 proxsuite::linalg::sparse::from_raw_parts,
656 n_tot,
657 n_tot,
658 0,
659 ldlt.col_ptrs.ptr(),
660 ldlt.nnz_counts.ptr(),
661 ldlt.row_indices.ptr(),
662 ldlt.values.ptr(),
663 };
664
665 I const* perm = work.internal.ldlt.perm.ptr();
666 I const* perm_inv = work.internal.ldlt.perm_inv.ptr();
667
668 for (isize i = 0; i < n_tot; ++i) {
669 tmp[i] = rhs_e[isize(zx(perm[i]))];
670 }
671
672 proxsuite::linalg::sparse::dense_lsolve<T, I>(
673 { proxsuite::linalg::sparse::from_eigen, tmp }, ld);
674 for (isize i = 0; i < n_tot; ++i) {
675 tmp[i] /= ld.values()[isize(zx(ld.col_ptrs()[i]))];
676 }
677 proxsuite::linalg::sparse::dense_ltsolve<T, I>(
678 { proxsuite::linalg::sparse::from_eigen, tmp }, ld);
679 for (isize i = 0; i < n_tot; ++i) {
680 rhs_e[i] = tmp[isize(zx(perm_inv[i]))];
681 }
682
683 } else {
684 tmp = work.internal.matrix_free_solver.get()->solve(rhs_e);
685 rhs_e = tmp;
686 }
687}

705 Settings<T> const& settings,
706 Workspace<T, I> const& work,
707 proxsuite::linalg::sparse::MatRef<T, I> const& kkt_active,
708 bool const* active_constraints,
709 isize n,
710 isize n_eq,
711 isize n_in,
712 proxsuite::linalg::veg::dynstack::DynStackMut stack)
713{
714 auto rhs_e = rhs.to_eigen();
715 auto sol_e = sol.to_eigen();
716
717 isize n_tot = n + n_eq + n_in;
718
719 if (init_guess.dim == n_tot) {
720 sol_e = init_guess.to_eigen();
721 } else {
722 sol_e.setZero();
723 }
724
725 LDLT_TEMP_VEC_UNINIT(T, err, n_tot, stack);
726 T prev_err_norm = std::numeric_limits<T>::infinity();
727
728 for (isize solve_iter = 0; solve_iter < settings.n_iterative_refinement;
729 ++solve_iter) {
730 auto err_x = err.head(n);
731 auto err_y = err.segment(n, n_eq);
732 auto err_z = err.tail(n_in);
733
734 auto sol_x = sol_e.head(n);
735 auto sol_y = sol_e.segment(n, n_eq);
736 auto sol_z = sol_e.tail(n_in);
737
738 err -= rhs_e;
739 if (solve_iter > 0) {
740 detail::noalias_symhiv_add(err, kkt_active.to_eigen(), sol_e);
741 err_x += state.rho * sol_x;
742 err_y += -state.mu_eq * sol_y;
743 for (isize i = 0; i < n_in; ++i) {
744 err_z[i] += (active_constraints[i] ? -state.mu_in : T(1)) * sol_z[i];
745 }
746 }
747
748 T err_norm = dense::infty_norm(err);
749 if (err_norm > prev_err_norm / T(2.0)) {
750 break;
751 }
752
753 prev_err_norm = err_norm;
754 detail::ldl_solve_in_place({ proxqp::from_eigen, err }, work, stack);
755
756 sol -= err;
757 }

867

872 ldlt.etree.resize_for_overwrite(n_tot);
873 ldlt.perm.resize_for_overwrite(n_tot);

881 n_tot,

920 detail::refactorize_req(proxsuite::linalg::veg::Tag<T>{},
921 proxsuite::linalg::veg::Tag<I>{},
922 n_tot,
923 H_nnz + A_nnz + C_nnz);
924
925 work.internal.storage.resize_for_overwrite(req.alloc_req());

927 // initial factorization
928 {
929 proxsuite::linalg::sparse::MatRef<T, I> kkt = kkt_storage.as_ref();

932
933 // H and A are always active

936 }
937
938 isize C_active_nnz = 0;
939 for (isize j = 0; j < n_in; ++j) {
940 auto is_active = results.active_constraints()[j];
941 kkt_nnz_counts[n + n_eq + j] = is_active
942 ? I(kkt.col_end(usize(n + n_eq + j)) -
943 kkt.col_start(usize(n + n_eq + j)))
944 : I(0);
945 C_active_nnz += kkt_nnz_counts[n + n_eq + j];
946 }
947
948 proxsuite::linalg::sparse::MatRef<T, I> kkt_active = {
949 proxsuite::linalg::sparse::from_raw_parts,
950 n_tot,
951 n_tot,
952 H_nnz + A_nnz + C_active_nnz,
953 kkt.col_ptrs(),
954 kkt_nnz_counts.data(),
955 kkt.row_indices(),
956 kkt.values(),
957 };
958

960 new typename Workspace<T, I>::MatrixFreeSolver);

962 new detail::AugmentedKkt<T, I>{ { kkt_active,
963 results.active_constraints(),
964 n,
965 n_eq,
966 n_in,
967 {},
968 {},
969 {} } });
970
971 isize ldlt_ntot = work.internal.do_ldlt ? n_tot : 0;
972 isize ldlt_lnnz = work.internal.do_ldlt ? max_lnnz : 0;

974 ldlt.nnz_counts.resize_for_overwrite(ldlt_ntot);
975 ldlt.row_indices.resize_for_overwrite(ldlt_lnnz);
976 ldlt.values.resize_for_overwrite(ldlt_lnnz);

979 proxsuite::linalg::sparse::from_raw_parts,
980 n_tot,
981 n_tot,
982 0,
983 ldlt.col_ptrs.ptr_mut(),
984 work.internal.do_ldlt ? ldlt.nnz_counts.ptr_mut() : nullptr,
985 ldlt.row_indices.ptr_mut(),
986 ldlt.values.ptr_mut(),
987 };
988
989 detail::refactorize(results.internal.state,
990 kkt_active,
991 results.active_constraints(),
992 work,
993 stack);
994 }

1112 rhs.tail(n_in).setZero();

1114 detail::ldl_iter_solve_in_place({ proxqp::from_eigen, rhs },
1115 { proxqp::from_eigen, no_guess },
1116 results.internal.state,
1117 settings,
1118 work,
1119 kkt_active,
1120 results.active_constraints(),
1121 n,
1122 n_eq,
1123 n_in,
1124 stack);
1125

1127 y = rhs.segment(n, n_eq);
1128 z = rhs.tail(n_in);
1129
1130 break;
1131 }
1132 default: {
1133 VEG_UNIMPLEMENTED();

1135 }
1136 }
1137 }
1138
1139 for (isize iter_outer = 0; iter_outer < settings.max_iter_outer;
1140 ++iter_outer) {
