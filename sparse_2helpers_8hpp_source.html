26template<typename T, typename I>
27void
28update_proximal_parameters(Results<T>& results,
29 Workspace<T, I>& work,
30 std::optional<T> rho_new,
31 std::optional<T> mu_eq_new,
32 std::optional<T> mu_in_new)
33{
34 if (rho_new != std::nullopt) {
35 results.info.rho = rho_new.value();
36 work.internal.proximal_parameter_update = true;
37 }
38 if (mu_eq_new != std::nullopt) {
39 results.info.mu_eq = mu_eq_new.value();
40 results.info.mu_eq_inv = T(1) / results.info.mu_eq;
41 work.internal.proximal_parameter_update = true;
42 }
43 if (mu_in_new != std::nullopt) {
44 results.info.mu_in = mu_in_new.value();
45 results.info.mu_in_inv = T(1) / results.info.mu_in;
46 work.internal.proximal_parameter_update = true;
47 }
48}
58template<typename T>
59void
60warm_start(std::optional<VecRef<T>> x_wm,
61 std::optional<VecRef<T>> y_wm,
62 std::optional<VecRef<T>> z_wm,
63 Results<T>& results,
64 Settings<T>& settings)
65{
66
67 isize n_eq = results.y.rows();
68 isize n_in = results.z.rows();
69 if (n_eq != 0) {
70 if (n_in != 0) {
71 if (x_wm != std::nullopt && y_wm != std::nullopt &&
72 z_wm != std::nullopt) {
73 results.x = x_wm.value().eval();
74 results.y = y_wm.value().eval();
75 results.z = z_wm.value().eval();
76 settings.initial_guess = InitialGuessStatus::WARM_START;
77 }
78 } else {
79 // n_in= 0
80 if (x_wm != std::nullopt && y_wm != std::nullopt) {
81 results.x = x_wm.value().eval();
82 results.y = y_wm.value().eval();
83 settings.initial_guess = InitialGuessStatus::WARM_START;
84 }
85 }
86 } else if (n_in != 0) {
87 // n_eq = 0
88 if (x_wm != std::nullopt && z_wm != std::nullopt) {
89 results.x = x_wm.value().eval();
90 results.z = z_wm.value().eval();
91 settings.initial_guess = InitialGuessStatus::WARM_START;
92 }
93 } else {
94 // n_eq = 0 and n_in = 0
95 if (x_wm != std::nullopt) {
96 results.x = x_wm.value().eval();
97 settings.initial_guess = InitialGuessStatus::WARM_START;
98 }
99 }

115template<typename T, typename I, typename P>
116void
117qp_setup(QpView<T, I> qp,
118 Results<T>& results,
119 Model<T, I>& data,
120 Workspace<T, I>& work,
121 Settings<T>& settings,
122 P& precond,
123 PreconditionerStatus& preconditioner_status)
124{
125 isize n = qp.H.nrows();
126 isize n_eq = qp.AT.ncols();
127 isize n_in = qp.CT.ncols();
128
129 if (results.x.rows() != n) {
130 results.x.resize(n);
131 results.x.setZero();
132 }
133 if (results.y.rows() != n_eq) {
134 results.y.resize(n_eq);
135 results.y.setZero();
136 }
137 if (results.z.rows() != n_in) {
138 results.z.resize(n_in);
139 results.z.setZero();
140 }
141 if (results.active_constraints.len() != n_in) {
142 results.active_constraints.resize(n_in);
143 for (isize i = 0; i < n_in; ++i) {
144 results.active_constraints[i] = false;
145 }
146 }
147 bool execute_preconditioner_or_not = false;
148 switch (preconditioner_status) {
149 case PreconditionerStatus::EXECUTE:
150 execute_preconditioner_or_not = true;
151 break;
152 case PreconditionerStatus::IDENTITY:
153 execute_preconditioner_or_not = false;
154 break;
155 case PreconditionerStatus::KEEP:
156 // keep previous one
157 execute_preconditioner_or_not = false;
158 break;
159 }
160 // performs scaling according to options chosen + stored model value
161 work.setup_impl(
162 qp,
163 results,
164 data,
165 settings,
166 execute_preconditioner_or_not,
167 precond,
168 P::scale_qp_in_place_req(proxsuite::linalg::veg::Tag<T>{}, n, n_eq, n_in));
169 switch (settings.initial_guess) { // the following is used when initiliazing
170 // the Qp object or updating it
171 case InitialGuessStatus::EQUALITY_CONSTRAINED_INITIAL_GUESS: {
172
173 if (work.internal.proximal_parameter_update) {
174 results.cleanup_all_except_prox_parameters();
175 } else {
176 results.cleanup();
177 }
178 break;
179 }
180 case InitialGuessStatus::COLD_START_WITH_PREVIOUS_RESULT: {
181 // keep solutions but restart workspace and results
182
183 if (work.internal.proximal_parameter_update) {
184 results.cleanup_statistics();
185 } else {
186 results.cold_start();
187 }
188 break;
189 }
190 case InitialGuessStatus::NO_INITIAL_GUESS: {
191
192 if (work.internal.proximal_parameter_update) {
193 results.cleanup_all_except_prox_parameters();
194 } else {
195 results.cleanup();
196 }
197 break;
198 }
199 case InitialGuessStatus::WARM_START: {
200
201 if (work.internal.proximal_parameter_update) {
202 results.cleanup_all_except_prox_parameters();
203 } else {
204 results.cleanup();
205 }
206 break;
207 }
208 case InitialGuessStatus::WARM_START_WITH_PREVIOUS_RESULT: {
209 // keep workspace and results solutions except statistics
210
211 results.cleanup_statistics(); // always keep prox parameters (changed or
212 // previous ones)
213 break;
214 }
215 }

223template<typename T, typename I>
224auto
225have_same_structure(proxsuite::linalg::sparse::MatRef<T, I> a,
226 proxsuite::linalg::sparse::MatRef<T, I> b) -> bool
227{
228 if (a.nrows() != b.nrows())
229 return false;
230 if (a.ncols() != b.ncols())
231 return false;
232 for (isize j = 0; j < a.ncols(); ++j) {
233 isize n_elems = a.col_end(j) - a.col_start(j);
234 if (n_elems != b.col_end(j) - b.col_start(j))
235 return false;
236 for (isize p = 0; p < n_elems; ++p) {
237 isize i_a = a.row_indices()[a.col_start(j) + p];
238 isize i_b = b.row_indices()[b.col_start(j) + p];
239 if (i_a != i_b)
240 return false;
241 }
242 }
243 return true;
244}
251template<typename T, typename I>
252void
253copy(proxsuite::linalg::sparse::MatMut<T, I> a,
254 proxsuite::linalg::sparse::MatRef<T, I> b)
255{
256 // assume same sparsity structure for a and b
257 // copy b into a
258 for (isize j = 0; j < a.ncols(); ++j) {
259 auto a_start = a.values_mut() + a.col_start(j);
260 auto b_start = b.values() + b.col_start(j);
261
262 auto n_elems = a.col_end(j) - a.col_start(j);
263
264 for (isize p = 0; p < n_elems; ++p) {
265 a_start[p] = b_start[p];
266 }
267 }
268}
