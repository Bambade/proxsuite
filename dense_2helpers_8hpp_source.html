33template<typename T>
34void
35compute_equality_constrained_initial_guess(Workspace<T>& qpwork,
36 const Settings<T>& qpsettings,
37 const Model<T>& qpmodel,
38 Results<T>& qpresults)
39{
40
41 qpwork.rhs.setZero();
42 qpwork.rhs.head(qpmodel.dim) = -qpwork.g_scaled;
43 qpwork.rhs.segment(qpmodel.dim, qpmodel.n_eq) = qpwork.b_scaled;
44 iterative_solve_with_permut_fact( //
45 qpsettings,
46 qpmodel,
47 qpresults,
48 qpwork,
49 T(1),
50 qpmodel.dim + qpmodel.n_eq);
51
52 qpresults.x = qpwork.dw_aug.head(qpmodel.dim);
53 qpresults.y = qpwork.dw_aug.segment(qpmodel.dim, qpmodel.n_eq);
54 qpwork.dw_aug.setZero();
55 qpwork.rhs.setZero();
56}

67template<typename T>
68void
69setup_factorization(Workspace<T>& qpwork,
70 const Model<T>& qpmodel,
71 Results<T>& qpresults)
72{
73
74 proxsuite::linalg::veg::dynstack::DynStackMut stack{
75 proxsuite::linalg::veg::from_slice_mut,
76 qpwork.ldl_stack.as_mut(),
77 };
78
79 qpwork.kkt.topLeftCorner(qpmodel.dim, qpmodel.dim) = qpwork.H_scaled;
80 qpwork.kkt.topLeftCorner(qpmodel.dim, qpmodel.dim).diagonal().array() +=
81 qpresults.info.rho;
82 qpwork.kkt.block(0, qpmodel.dim, qpmodel.dim, qpmodel.n_eq) =
83 qpwork.A_scaled.transpose();
84 qpwork.kkt.block(qpmodel.dim, 0, qpmodel.n_eq, qpmodel.dim) = qpwork.A_scaled;
85 qpwork.kkt.bottomRightCorner(qpmodel.n_eq, qpmodel.n_eq).setZero();
86 qpwork.kkt.diagonal()
87 .segment(qpmodel.dim, qpmodel.n_eq)
88 .setConstant(-qpresults.info.mu_eq);
89
90 qpwork.ldl.factorize(qpwork.kkt, stack);
91}

108 preconditioner::RuizEquilibration<T>& ruiz,

111

267template<typename Mat, typename T>
268void
269setup( //
270 Mat const& H,
271 VecRef<T> g,
272 Mat const& A,
273 VecRef<T> b,
274 Mat const& C,
275 VecRef<T> u,
276 VecRef<T> l,
277 Settings<T>& qpsettings,
278 Model<T>& qpmodel,
279 Workspace<T>& qpwork,
280 Results<T>& qpresults,
281 preconditioner::RuizEquilibration<T>& ruiz,
282 PreconditionerStatus preconditioner_status)
283{
284
285 switch (qpsettings.initial_guess) {
286 case InitialGuessStatus::EQUALITY_CONSTRAINED_INITIAL_GUESS: {
287 if (qpwork.proximal_parameter_update) {
288 qpresults.cleanup_all_except_prox_parameters();
289 } else {
290 qpresults.cleanup();
291 }
292 qpwork.cleanup();
293 break;
294 }
295 case InitialGuessStatus::COLD_START_WITH_PREVIOUS_RESULT: {
296 // keep solutions but restart workspace and results
297 if (qpwork.proximal_parameter_update) {
298 qpresults.cleanup_statistics();
299 } else {
300 qpresults.cold_start();
301 }
302 qpwork.cleanup();
303 break;
304 }
305 case InitialGuessStatus::NO_INITIAL_GUESS: {
306 if (qpwork.proximal_parameter_update) {
307 qpresults.cleanup_all_except_prox_parameters();
308 } else {
309 qpresults.cleanup();
310 }
311 qpwork.cleanup();
312 break;
313 }
314 case InitialGuessStatus::WARM_START: {
315 if (qpwork.proximal_parameter_update) {
316 qpresults
317 .cleanup_all_except_prox_parameters(); // the warm start is given at
318 // the solve function
319 } else {
320 qpresults.cleanup();
321 }
322 qpwork.cleanup();
323 break;
324 }
325 case InitialGuessStatus::WARM_START_WITH_PREVIOUS_RESULT: {
326 if (qpwork.refactorize || qpwork.proximal_parameter_update) {
327 qpwork.cleanup(); // meaningful for when there is an upate of the model
328 // and one wants to warm start with previous result
329 qpwork.refactorize = true;
330 }
331 qpresults.cleanup_statistics();
332 break;
333 }
334 }
335 qpmodel.H =
336 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic, to_eigen_layout(rowmajor)>(
337 H);
338 qpmodel.g = g;
339 qpmodel.A =
340 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic, to_eigen_layout(rowmajor)>(
341 A);
342 qpmodel.b = b;
343 qpmodel.C =
344 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic, to_eigen_layout(rowmajor)>(
345 C);
346 qpmodel.u = u;
347 qpmodel.l = l;
348
349 qpwork.H_scaled = qpmodel.H;
350 qpwork.g_scaled = qpmodel.g;
351 qpwork.A_scaled = qpmodel.A;
352 qpwork.b_scaled = qpmodel.b;
353 qpwork.C_scaled = qpmodel.C;
354 qpwork.u_scaled = qpmodel.u;
355 qpwork.l_scaled = qpmodel.l;
356
357 qpwork.primal_feasibility_rhs_1_eq = infty_norm(qpmodel.b);
358 qpwork.primal_feasibility_rhs_1_in_u = infty_norm(qpmodel.u);
359 qpwork.primal_feasibility_rhs_1_in_l = infty_norm(qpmodel.l);
360 qpwork.dual_feasibility_rhs_2 = infty_norm(qpmodel.g);
361
362 switch (preconditioner_status) {
363 case PreconditionerStatus::EXECUTE:
364 setup_equilibration(qpwork, qpsettings, ruiz, true);
365 break;
366 case PreconditionerStatus::IDENTITY:
367 setup_equilibration(qpwork, qpsettings, ruiz, false);
368 break;
369 case PreconditionerStatus::KEEP:
370 // keep previous one
371 setup_equilibration(qpwork, qpsettings, ruiz, false);
372 break;
373 }
374}

386template<typename T>
387void
388update_proximal_parameters(Results<T>& results,
389 Workspace<T>& work,
390 std::optional<T> rho_new,
391 std::optional<T> mu_eq_new,
392 std::optional<T> mu_in_new)
393{
394
395 if (rho_new != std::nullopt) {
396 results.info.rho = rho_new.value();
397 work.proximal_parameter_update = true;
398 }
399 if (mu_eq_new != std::nullopt) {
400 results.info.mu_eq = mu_eq_new.value();
401 results.info.mu_eq_inv = T(1) / results.info.mu_eq;
402 work.proximal_parameter_update = true;
403 }
404 if (mu_in_new != std::nullopt) {
405 results.info.mu_in = mu_in_new.value();
406 results.info.mu_in_inv = T(1) / results.info.mu_in;
407 work.proximal_parameter_update = true;
408 }

419template<typename T>
420void
421warm_start(std::optional<VecRef<T>> x_wm,
422 std::optional<VecRef<T>> y_wm,
423 std::optional<VecRef<T>> z_wm,
424 Results<T>& results,
425 Settings<T>& settings)
426{
427
428 isize n_eq = results.y.rows();
429 isize n_in = results.z.rows();
430 if (n_eq != 0) {
431 if (n_in != 0) {
432 if (x_wm != std::nullopt && y_wm != std::nullopt &&
433 z_wm != std::nullopt) {
434 results.x = x_wm.value().eval();
435 results.y = y_wm.value().eval();
436 results.z = z_wm.value().eval();
437 }
438 } else {
439 // n_in= 0
440 if (x_wm != std::nullopt && y_wm != std::nullopt) {
441 results.x = x_wm.value().eval();
442 results.y = y_wm.value().eval();
443 }
444 }
445 } else if (n_in != 0) {
446 // n_eq = 0
447 if (x_wm != std::nullopt && z_wm != std::nullopt) {
448 results.x = x_wm.value().eval();
449 results.z = z_wm.value().eval();
450 }
451 } else {
452 // n_eq = 0 and n_in = 0
453 if (x_wm != std::nullopt) {
454 results.x = x_wm.value().eval();
455 }
456 }
457
458 settings.initial_guess = InitialGuessStatus::WARM_START;
459}
